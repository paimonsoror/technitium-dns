Thank you for installing {{ .Chart.Name }}!

Your Technitium DNS Server has been deployed to your Kubernetes cluster.

To get started:

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.web.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "technitium-dns-server.fullname" . }}-web)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Web Console: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.web.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "technitium-dns-server.fullname" . }}-web'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "technitium-dns-server.fullname" . }}-web --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Web Console: http://$SERVICE_IP:{{ .Values.service.web.port }}"
{{- else if contains "ClusterIP" .Values.service.web.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "technitium-dns-server.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Web Console: http://127.0.0.1:8080"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Get the DNS service endpoint:
{{- if contains "LoadBalancer" .Values.service.dns.type }}
  export DNS_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "technitium-dns-server.fullname" . }}-dns --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "DNS Server IP: $DNS_IP"
{{- else if contains "NodePort" .Values.service.dns.type }}
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "DNS Server IP: $NODE_IP"
{{- else }}
  echo "DNS Server: {{ include "technitium-dns-server.fullname" . }}-dns.{{ .Release.Namespace }}.svc.cluster.local"
{{- end }}

3. Login credentials:
{{- if .Values.dnsServer.adminPassword }}
  Username: admin
  Password: {{ .Values.dnsServer.adminPassword }}
{{- else }}
  Username: admin
  Password: Check the container logs for the auto-generated password:
    kubectl logs --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "technitium-dns-server.name" . }}" | grep "DNS Server was initialized"
{{- end }}

{{- end }}
